    @axiom
def eq_ref(s: Node, p: Node, o: Node):
    """
    EQ REF axiom

    :param s:
    :param p:
    :param o:
    :return:
    """
    assert Triple(s, p, o) >> And(
        Triple(s, OWL_SAME_AS, s),
        Triple(p, OWL_SAME_AS, p),
        Triple(o, OWL_SAME_AS, o),
    )
